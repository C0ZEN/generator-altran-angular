(function (angular) {
	'use strict';

	angular
		.module('<%= appNameCamel %>')
		.config(provider);

		'altranWorkflowItemProvider',
		'config'
	];

	function provider(altranWorkflowItemProvider, config) {

			.defaultSize(config.directives.workflow.item.defaultSize)

			.labels(config.directives.workflow.item.labels);
	}

})(window.angular);
