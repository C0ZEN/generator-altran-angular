(function (angular) {
	'use strict';

	angular
		.module('<%= appNameCamel %>')
		.config(provider);

	provider.$inject = [
		'altranBreadcrumbProvider',
		'config'
	];

	function provider(altranBreadcrumbProvider, config) {
		altranBreadcrumbProvider
			.enableLog(config.directives.breadcrumb.debug)
			.defaultSize(config.directives.breadcrumb.defaultSize);
	}

})(window.angular);
