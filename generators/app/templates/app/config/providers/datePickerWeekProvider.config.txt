(function (angular) {
	'use strict';

	angular
		.module('<%= appNameCamel %>')
		.config(provider);

	provider.$inject = [
		'altranDatePickerWeekProvider',
		'config'
	];

	function provider(altranDatePickerWeekProvider, config) {
		altranDatePickerWeekProvider
			.visible(config.directives.form.datePicker.week.visible)
			.format(config.directives.form.datePicker.week.format);
	}

})(window.angular);
